Workflow Examples

Real-world workflow examples to help you get started quickly with ForgePulse.

1. User Onboarding Flow

Automated email sequence for new user onboarding with delayed follow-ups.

Use Case

Send a welcome email immediately after user registration, wait 24 hours, then send a follow-up email with tips and resources.

use AlizHarb\ForgePulse\Models\Workflow;
use AlizHarb\ForgePulse\Enums\{WorkflowStatus, StepType};

// Create workflow
$workflow = Workflow::create([
    'name' => 'User Onboarding',
    'description' => 'Automated onboarding sequence',
    'status' => WorkflowStatus::ACTIVE,
]);

// Step 1: Send welcome email
$workflow->steps()->create([
    'name' => 'Send Welcome Email',
    'type' => StepType::NOTIFICATION,
    'position' => 1,
    'configuration' => [
        'notification_class' => \App\Notifications\WelcomeEmail::class,
        'recipients' => ['{{user_id}}'],
    ],
]);

// Step 2: Wait 24 hours
$workflow->steps()->create([
    'name' => 'Wait 24 Hours',
    'type' => StepType::DELAY,
    'position' => 2,
    'configuration' => [
        'seconds' => 86400,
    ],
]);

// Step 3: Send follow-up email
$workflow->steps()->create([
    'name' => 'Send Follow-up Email',
    'type' => StepType::NOTIFICATION,
    'position' => 3,
    'configuration' => [
        'notification_class' => \App\Notifications\FollowUpEmail::class,
        'recipients' => ['{{user_id}}'],
    ],
]);

// Execute workflow
$execution = $workflow->execute([
    'user_id' => $user->id,
    'email' => $user->email,
]);

2. Order Processing Workflow

Complete order processing with inventory check, payment, and notifications.

Pro Tip: Use conditional logic to handle different order scenarios like out-of-stock items or payment failures. 
$workflow = Workflow::create([
    'name' => 'Order Processing',
    'status' => WorkflowStatus::ACTIVE,
]);

// Step 1: Validate order
$validateStep = $workflow->steps()->create([
    'name' => 'Validate Order',
    'type' => StepType::ACTION,
    'position' => 1,
    'configuration' => [
        'action_class' => \App\Actions\ValidateOrderAction::class,
        'parameters' => ['order_id' => '{{order_id}}'],
    ],
]);

// Step 2: Check inventory (conditional)
$inventoryStep = $workflow->steps()->create([
    'name' => 'Check Inventory',
    'type' => StepType::ACTION,
    'position' => 2,
    'parent_step_id' => $validateStep->id,
    'configuration' => [
        'action_class' => \App\Actions\CheckInventoryAction::class,
        'parameters' => ['order_id' => '{{order_id}}'],
    ],
    'conditions' => [
        'operator' => 'and',
        'rules' => [
            ['field' => 'order.status', 'operator' => '==', 'value' => 'valid'],
        ],
    ],
]);

// Step 3: Process payment
$paymentStep = $workflow->steps()->create([
    'name' => 'Process Payment',
    'type' => StepType::ACTION,
    'position' => 3,
    'parent_step_id' => $inventoryStep->id,
    'configuration' => [
        'action_class' => \App\Actions\ProcessPaymentAction::class,
        'parameters' => ['order_id' => '{{order_id}}'],
    ],
    'conditions' => [
        'operator' => 'and',
        'rules' => [
            ['field' => 'inventory.available', 'operator' => '==', 'value' => true],
        ],
    ],
]);

// Step 4: Send confirmation
$workflow->steps()->create([
    'name' => 'Send Confirmation',
    'type' => StepType::NOTIFICATION,
    'position' => 4,
    'parent_step_id' => $paymentStep->id,
    'configuration' => [
        'notification_class' => \App\Notifications\OrderConfirmation::class,
        'recipients' => ['{{user_id}}'],
    ],
]);

3. Multi-Level Approval Workflow

Expense approval process with conditional routing based on amount.

$workflow = Workflow::create([
    'name' => 'Expense Approval',
    'status' => WorkflowStatus::ACTIVE,
]);

// Step 1: Manager approval
$managerApproval = $workflow->steps()->create([
    'name' => 'Manager Approval',
    'type' => StepType::NOTIFICATION,
    'position' => 1,
    'configuration' => [
        'notification_class' => \App\Notifications\ApprovalRequest::class,
        'recipients' => ['{{manager_id}}'],
    ],
]);

// Step 2: Director approval (for amounts > $1000)
$directorApproval = $workflow->steps()->create([
    'name' => 'Director Approval',
    'type' => StepType::NOTIFICATION,
    'position' => 2,
    'parent_step_id' => $managerApproval->id,
    'configuration' => [
        'notification_class' => \App\Notifications\ApprovalRequest::class,
        'recipients' => ['{{director_id}}'],
    ],
    'conditions' => [
        'operator' => 'and',
        'rules' => [
            ['field' => 'amount', 'operator' => '>', 'value' => 1000],
            ['field' => 'manager_approved', 'operator' => '==', 'value' => true],
        ],
    ],
]);

// Step 3: Process expense
$workflow->steps()->create([
    'name' => 'Process Expense',
    'type' => StepType::ACTION,
    'position' => 3,
    'parent_step_id' => $directorApproval->id,
    'configuration' => [
        'action_class' => \App\Actions\ProcessExpenseAction::class,
        'parameters' => ['expense_id' => '{{expense_id}}'],
    ],
]);

4. Webhook Integration

Integrate with external services using webhooks.

$workflow = Workflow::create([
    'name' => 'Slack Notification Workflow',
    'status' => WorkflowStatus::ACTIVE,
]);

// Step 1: Send to Slack
$workflow->steps()->create([
    'name' => 'Send Slack Message',
    'type' => StepType::WEBHOOK,
    'position' => 1,
    'configuration' => [
        'url' => 'https://hooks.slack.com/services/YOUR/WEBHOOK/URL',
        'method' => 'POST',
        'headers' => [
            'Content-Type' => 'application/json',
        ],
        'payload' => [
            'text' => 'New order received: {{order_id}}',
            'channel' => '#orders',
            'username' => 'ForgePulse Bot',
        ],
    ],
]);

// Step 2: Log to external analytics
$workflow->steps()->create([
    'name' => 'Log to Analytics',
    'type' => StepType::WEBHOOK,
    'position' => 2,
    'configuration' => [
        'url' => 'https://api.analytics.com/events',
        'method' => 'POST',
        'headers' => [
            'Authorization' => 'Bearer {{api_token}}',
            'Content-Type' => 'application/json',
        ],
        'payload' => [
            'event' => 'order_created',
            'properties' => [
                'order_id' => '{{order_id}}',
                'amount' => '{{amount}}',
            ],
        ],
    ],
]);

6. Queued Job Workflow

Dispatch Laravel jobs with custom queues and delays.

$workflow = Workflow::create([
    'name' => 'Data Processing Pipeline',
    'status' => WorkflowStatus::ACTIVE,
]);

// Step 1: Import data (high priority queue)
$workflow->steps()->create([
    'name' => 'Import Data',
    'type' => StepType::JOB,
    'position' => 1,
    'configuration' => [
        'job_class' => \App\Jobs\ImportDataJob::class,
        'parameters' => ['file_path' => '{{file_path}}'],
        'queue' => 'high-priority',
        'delay' => null,
    ],
]);

// Step 2: Process data (delayed 5 minutes)
$workflow->steps()->create([
    'name' => 'Process Data',
    'type' => StepType::JOB,
    'position' => 2,
    'configuration' => [
        'job_class' => \App\Jobs\ProcessDataJob::class,
        'parameters' => ['batch_id' => '{{batch_id}}'],
        'queue' => 'default',
        'delay' => 300, // 5 minutes
    ],
]);

7. Complex Conditional Logic

Advanced conditional branching with nested AND/OR logic.

$workflow = Workflow::create([
    'name' => 'Smart Routing Workflow',
    'status' => WorkflowStatus::ACTIVE,
]);

$rootStep = $workflow->steps()->create([
    'name' => 'Initial Check',
    'type' => StepType::ACTION,
    'position' => 1,
    'configuration' => [
        'action_class' => \App\Actions\InitialCheckAction::class,
    ],
]);

// Premium user path
$workflow->steps()->create([
    'name' => 'Premium User Handler',
    'type' => StepType::ACTION,
    'position' => 2,
    'parent_step_id' => $rootStep->id,
    'configuration' => [
        'action_class' => \App\Actions\PremiumUserAction::class,
    ],
    'conditions' => [
        'operator' => 'and',
        'rules' => [
            ['field' => 'user.subscription', 'operator' => '==', 'value' => 'premium'],
            [
                'operator' => 'or',
                'rules' => [
                    ['field' => 'user.lifetime_value', 'operator' => '>', 'value' => 1000],
                    ['field' => 'user.referrals', 'operator' => '>=', 'value' => 5],
                ],
            ],
        ],
    ],
]);

8. Using Workflow Templates

Save and reuse workflow configurations.

// Create a workflow
$workflow = Workflow::create([
    'name' => 'Customer Onboarding v1',
    'status' => WorkflowStatus::ACTIVE,
]);

// Add steps...
$workflow->steps()->create([...]);

// Save as template
$template = $workflow->saveAsTemplate('Customer Onboarding Template');

// Later, create new workflow from template
$newWorkflow = $template->instantiateFromTemplate('Customer Onboarding v2');

// Export template to file
$templateManager = app(\AlizHarb\ForgePulse\Services\TemplateManager::class);
$path = $templateManager->export($template);

// Import template from file
$importedWorkflow = $templateManager->import($path, 'Imported Workflow');

9. Real-Time Execution Tracking

Monitor workflow execution in real-time with Livewire.

{{-- In your Blade view --}}
<div>
    <h2>Workflow Execution</h2>
    
    {{-- Real-time execution tracker --}}
    <livewire:forgepulse::workflow-execution-tracker :execution="$execution" />
    
    {{-- Or use the workflow builder --}}
    <livewire:forgepulse::workflow-builder :workflow="$workflow" />
</div>
// In your controller
public function show(Workflow $workflow)
{
    $execution = $workflow->execute([
        'user_id' => auth()->id(),
        'data' => request('data'),
    ]);
    
    return view('workflows.show', [
        'workflow' => $workflow,
        'execution' => $execution,
    ]);
}

10. Listening to Workflow Events

React to workflow events in your application.

// In EventServiceProvider.php
use AlizHarb\ForgePulse\Events\{
    WorkflowStarted,
    WorkflowCompleted,
    WorkflowFailed,
    StepExecuted
};

protected $listen = [
    WorkflowStarted::class => [
        \App\Listeners\LogWorkflowStart::class,
        \App\Listeners\NotifyAdministrators::class,
    ],
    
    WorkflowCompleted::class => [
        \App\Listeners\LogWorkflowCompletion::class,
        \App\Listeners\SendCompletionNotification::class,
    ],
    
    WorkflowFailed::class => [
        \App\Listeners\LogWorkflowFailure::class,
        \App\Listeners\NotifyDevelopers::class,
    ],
];
// Example listener
namespace App\Listeners;

use AlizHarb\ForgePulse\Events\WorkflowCompleted;
use App\Notifications\WorkflowCompletedNotification;

class SendCompletionNotification
{
    public function handle(WorkflowCompleted $event): void
    {
        $execution = $event->execution;
        $user = $execution->user;
        
        $user->notify(new WorkflowCompletedNotification($execution));
    }
}
